121      * This method manages master / slaves connections
122      * @param string name, the name of the instance
123      * @param DB the connected instance

125     public static function get( $name){
126         if(isset(self::$instances[$name])){
127             return self::$instances[$name];
128         }
129 
130         $servers = self::$servers[$name];
131         foreach($servers as $i => $server){
132             try{
133                 self::$instances[$name] = new self($server);
134                 App::logger()->debug('Connection to db ' . $name . ' successfull');
135                 // The connection succeed
136                 break;
137             }
138             catch(DBException $e){
139                 App::logger()->notice('Impossible to connect to db ' . $name . ' on instance ' . $i . ' : ' . $e->getMessage());
140                 // The connection failed, try to connect to the next slave
141                 if(!isset($servers[$i+1])){
142                     // the last slave connection failed
143                     App::logger()->error('Impossible to connect to db ' . $name . ' : ' . $e->getMessage());
144                     throw $e;
145                 }
146             }
147         }

152     /**
153      * Execute a SQL query on the SQL server, and get the result of execution
154      * @param $sql The query to execute,
155      * @param array $binds The binded value to send to the server
156      * @param array $options The result options. This array can contains the following data :
157      *                      - 'return' (mixed) : The type of return (default DB::RETURN_STATUS). It can take the following values :
158      *                          . DB::RETURN_STATUS : Returns the execution status
159      *                          . DB::RETURN_ARRAY : Each result row is an associative array, with columns as keys
160      *                          . DB::RETURN_OBJECT : Eeach result row is a StdClass instance, with columns as properties
161      *                          . DB::RETURN_LAST_INSERT_ID : Return the last inserted id
162      *                          . DB::RETURN_AFFECTED_ROWS : Returns the number of affected rows
163      *                          . DB::RETURN_QUERY : Returns the query itself
164      *                          . DB::RETURN_CURSOR : Returns a cursor to fetch the results
165      *                          . A classname : Each result row is an instance of the given classname, with columns as properties
166      *                      - 'onerow' (bool) : If true, the function will return the first row of the result set (default false)
167      *                      - 'index' (string) : Defines the column values to get as array result index (default null)
168      *                      - 'args' (array) : This can be used when you define a classname as 'return', to pass arguments to the class constructor
169      * @return mixed The execution result, depending on what type of return has been defined in $options parameter
170      */
171     public function query($sql, $binds = array(), $options = array()){
172         $default = array(
173             'return' => self::RETURN_STATUS,
174             'onerow' => false,
175             'index' => null,
176             'args' => array()
177         );
178 
179         foreach($default as $name => $value){
180             if(!isset($options[$name])){
181                 $options[$name] = $value;
182             }
183         }
184 
185         try {
186             // Prepare the query
187             $stmt = $this->connection->prepare($sql);
188 
189             // Prepare query logging
190             $log = $sql;
191 
192             // Bind values to the query
193             foreach($binds as $key => $bind){
194                 $stmt->bindValue(":$key", $bind);
195                 $log = str_replace(":$key", $this->connection->quote($bind), $log);
196             }
197 
198             // Execute the query
199             $start = microtime(true);
200             $stmt->execute();
201             $end = microtime(true);
202 
203             // Get the result
204             $result;
205             if(is_int($options['return'])){
206                 switch($options['return']){
207                     // Return the query status
208                     case self::RETURN_STATUS:
209                         $result =  true;
210                     break;
211 
212                     case self::RETURN_ARRAY:
213                         if($options['onerow']){
214                             // Return the first row as associative array
215                             $result = $stmt->fetch(\PDO::FETCH_ASSOC);
216                         }
217                         else{
218                             // Return an array of associative arrays
219                             $data = array();
220                             while($row = $stmt->fetch(\PDO::FETCH_ASSOC)){
221                                 if($options['index']){
222                                     $data[$row[$options['index']]] = $row;
223                                 }
224                                 else{
225                                     $data[]= $row;
226                                 }
227                             }
228                             $result = $data;
229                         }
230                     break;
231 
232                     case self::RETURN_OBJECT :
233                         if($options['onerow']){
234                             // Return the first row as an StdClass instance
235                             $result = $stmt->fetch(\PDO::FETCH_OBJ);
236                         }
237                         else{
238                             // Return an array of StdClass instances
239                             $data = array();
240                             while($row = $stmt->fetch(\PDO::FETCH_OBJ)){
241                                 if($options['index']){
242                                     $index = $options['index'];
243                                     $data[$row->$index]= $row;
244                                 }
245                                 else{
246                                     $data[] = $row;
247                                 }
248                             }
249                             $result = $data;
250                         }
251                     break;
252 
253                     case self::RETURN_LAST_INSERT_ID :
254                         // Return the last inserted array
255                         $result = $this->connection->lastInsertId();
256                     break;
257 
258                     case self::RETURN_AFFECTED_ROWS :
259                         // Return the number of affected rows
260                         $result = $stmt->rowCount();
261                     break;
262 
263                     case self::RETURN_CURSOR :
264                         // Return a cursor to fetch the results
265                         $result = $stmt;
266                     break;
267                 }
268             }
269             else{
270                 if($options['onerow']){
271                     // Return a model instance
272                     $object = $stmt->fetchObject($options['return'], $options['args']);
273                     $result = $object !== false ? $object : null;
274                 }
275                 else{
276                     // Return an array of model instances
277                     $data = array();
278                     while($row = $stmt->fetchObject($options['return'], $options['args'])){
279                         if($options['index']){
280                             $index = $options['index'];
281                             $data[$row->$index] = $row;
282                         }
283                         else{
284                             $data[] = $row;
285                         }
286                     }
287                     $result = $data;
288                 }
289             }
290 
291             // Log the query
292             $this->addLog(str_replace(PHP_EOL, ' ', $log), $result, $start, $end);
293 
294             return $result;
295         }
296         catch(\PDOException $e) {
297             $this->addLog($log, 'query failed', $start, microtime(true));
298             throw new DBException($e->getMessage(), DBException::QUERY_ERROR, $sql);
299         }
300     }

313      * Build and execute a SELECT query on the selected database, and return the result
314      * @param array $query The parameters of the query. This parameter can have the following data :
315      *                      - 'from' string (required) : The table name where to search
316      *                      - 'fields' array (optionnal) : The fields to get. Each element of this array can be a single element of the column name, or a key/value combination that will be parse to 'key as value' in the SQL query
317      *                      - 'where' string | DBExample (optionnal) : The search condition. Can be a SQL expression or a DBExample instance
318      *                      - 'group' array (optionnal) : The columns to group the result, each column in an array element
319      *                      - 'having' string (optinnal) : The 'HAVING' expression, formatted as a SQL expression
320      *                      - 'orderby' array (optionnal) : The result sort, where each key is a column name, and the values define the order (ASC, DESC)
321      *                      - 'start' int (optionnal) : The first row number to get the results
322      *                      - 'limit' int (optionnal) : The maximum number of rows to return
323      *                      - 'one' bool (optionnal) : If set to true, then the first row will be returned
324      *                      - 'binds' array (optionnal) : The binded values
325      *                      - 'index' string (optionnal) : If set, the result arrar will be indexed by the value of the column set by this property
326      *                      - 'return' mixed (default : DB::RETURN_ARRAY) : The return type (all possible values are defined on the method 'query')
327      *  @return mixed The query result
328      */
329     public function select($query) {
330         $query = (object) $query;
331 
332         if(!isset($query->from)){
333             throw new DBException("'from' parameter is mandatory in select method", DBException::QUERY_ERROR, '' );
334         }
335 
336         /*** Treat the paramters ***/
337         if(empty($query->fields)){
338             $query->fields = "*";
339         }
340         elseif(is_array($query->fields)){
341             $tmp = array();
342             foreach($query->fields as $name => $alias){
343                 if(is_numeric($name)){
344                     $tmp[] = $alias;
345                 }
346                 else{
347                     $tmp[] = "$name as $alias";
348                 }
349             }
350             $query->fields = implode(",", $tmp);
351         }
352 
353         if(isset($query->where) && $query->where instanceof DBExample){
354             $query->where = $query->where->parse($query->binds);
355         }
356         $where = !empty($query->where) ? "WHERE $query->where" : '';
357 
358         $group = !empty($query->group) ? "GROUP BY ".implode(",", array_map(array($this, 'formatField'), $query->group)) : "";
359 
360         $having = !empty($query->having) ? "HAVING $query->having" : '';
361 
362         if(!empty($query->orderby)){
363             $orders = array();
364             foreach($query->orderby as $field => $value){
365                 $orders[] = self::formatField($field) . " " . $value;
366             }
367             $orderby = "ORDER BY " . implode(",", $orders);
368         }
369         else{
370             $orderby = "";
371         }
372 
373         $limit = "";
374         if(!empty($query->one)){
375             $limit = "LIMIT 1";
376         }
377         elseif(!empty($query->limit)){
378             $limit = " LIMIT ".(!empty($query->start) ? "$query->start, " : ""). "$query->limit";
379         }
380 
381         $sql = "SELECT $query->fields FROM $query->from $where $group $having $orderby $limit";
382 
383         if(empty($query->binds)){
384            $query->binds = array();
385         }
386 
387         if(empty($query->return)){
388             $query->return = self::RETURN_ARRAY;
389         }
390 
391         return $this->query($sql, $query->binds, array(
392             'return' => $query->return,
393             'onerow' => !empty($query->one),
394             'index' => empty($query->index) ? '' : $query->index
395         ));
396     }
