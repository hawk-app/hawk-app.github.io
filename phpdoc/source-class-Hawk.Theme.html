333     /**
334      * Build the theme : copy every resource files in themes/{themename}
335      * @param boolean $force If set to true, the theme will be rebuilt without condition
336      */
337     public function build($force = false){
338         if($this->getDefinition('extends')){
339             if(Theme::get($this->getDefinition('extends'))){
340                 Theme::get($this->getDefinition('extends'))->build($force);
341             }
342         }
343         $build = false;
344         if($force){
345             $build = true;
346         }
347 
348         if(!file_exists($this->getStaticDir())){
349             mkdir($this->getStaticDir(), 0755, true);
350             $build = true;
351         }
352 
353 
354         if(!$build){
355             $dest = $this->getStaticCssFile();
356 
357             if(!is_file($dest)){
358                 $build = true;
359             }
360             else{
361                 // Get all files in less/
362                 $files = App::fs()->find($this->getLessDirname(), '*.less');
363                 $lastUpdate = filemtime($dest);
364                 foreach($files as $file){
365                     if(filemtime($file) > $lastUpdate){
366                         $build = true;
367                         break;
368                     }
369                 }
370             }
371         }
372 
373         if($build){
374             // Build the theme => Copy each accessible files in static dir
375             foreach(glob($this->getRootDir() . '*' ) as $elt){
376                 if(! in_array(basename($elt), array('views', 'start.php'))) {
377                     App::fs()->copy($elt, $this->getStaticDir());
378                 }
379             }
380 
381             // In the main less file, replace the editable vars by their customized values
382             $values = $this->getVariablesCustomValues();
383             $precompiledLess = preg_replace_callback(self::EDITABLE_VARS_PATTERN, function($m) use($values){
384                 return '@' . $m[1] . ' : ' . (isset($values[$m[1]]) ? $values[$m[1]] : $m[2]) . ';';
385             }, file_get_contents($this->getBaseLessFile()));
386 
387             file_put_contents($this->getStaticLessFile(), $precompiledLess);
388 
389             Less::compile($this->getStaticLessFile(), $this->getStaticCssFile());
390         }
391 
392         return $build;
393     }

396     /**
397      * Get the variables in a CSS file content. In CSS files, variables are defined with the folowing format :
398      * /* define("variableName", color|dimension|file, "variable description that will appear in the customization page of the theme", defaultValue) *\/
399      * @param string $less The Less code to parse
400      * @return array The variables, where each element contains the 'name' of the variable, it 'type', it 'description', and it 'default' value
401      */
402     public function getEditableVariables($less = null){
403         if(!$less){
404             $less = file_get_contents($this->getBaseLessFile());
405         }
406         preg_match_all(self::EDITABLE_VARS_PATTERN, $less, $matches, PREG_SET_ORDER);
407         $variables = array();
408         foreach($matches as $match){
409 
410             preg_match_all('#^{a-z}{/a-z}$#', $match[3], $matches_description, PREG_SET_ORDER);
411 
412             App::logger()->error('match=');
413             $description = array();
414             foreach($matches_description as $match_description){
415                 App::logger()->error('match=' . $match_description[1] . "=" . $match_description[2]);
416                 $description[$match_description[1]] = $match_description[2];
417             }
418 
419             $variables[] = array(
420                 'name' => $match[1],
421                 'default' => $match[2],
422                 'description' => $match[3],
423                 'type' => isset($match[4]) ? $match[4] : ''
424             );
425         }
426         return $variables;
427     }
