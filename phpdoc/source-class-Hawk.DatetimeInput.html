 83      * @return bool the check status
 84      */
 85     public function check(&$form = null){
 86         // First check the global input validators
 87         if(! parent::check($form)){
 88             return false;
 89         }
 90         
 91         if($this->value!=""){               
 92             // Check the format of the given date
 93             $tmp = date_parse_from_format($this->format, $this->value);
 94             if(empty($tmp)){
 95                 $form->error($this->errorAt, Lang::get('form.date-format'));
 96                 return false;
 97             }       
 98             // Check the date is valid
 99             if(!checkdate($tmp['month'], $tmp['day'], $tmp['year'])){                   
100                 $form->error($this->errorAt, Lang::get('form.invalid-date'));
101                 return false;
102             }                       
103             
104         }           
105         return true;
106     }
