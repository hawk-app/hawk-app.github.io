 11 /**
 12  * This class has utilities to manage the file system, for missing functions in PHP
 13  * @package Utils
 14  */
 15 final class FileSystem extends Singleton{   
 16     /**
 17      * Directory separator
 18      */
 19     const DS = DIRECTORY_SEPARATOR;
 20 
 21     const FIND_FILE_ONLY = 'file';
 22     const FIND_DIR_ONLY = 'dir';
 23     const FIND_ANY_TYPE = 'any';
 24 
 25     /**
 26      * The filesystem instance
 27      */
 28     protected static $instance;

 31     /**
 32      * Get all files in a directory, including those beginning by '.'
 33      * @param string $dir The directory to find files into
 34      * @return array The list of files matching the pattern
 35      */
 36     public function getAllFilesInDir($dir){
 37         $files = array_merge(glob($dir . self::DS . '*'), glob($dir . self::DS . '.*'));
 38         return array_filter($files, function($file){
 39             return basename($file) != '.' && basename($file) != '..';
 40         });
 41     }
 42 
 43     /**
 44      * Equivalent to cp -r 
 45      * @param string $source The source file or directory to copy
 46      * @param string $dest The destination file or directory
 47      * @static
 48      */
 49     public function copy($source, $dest){
 50         if(basename($source) == '*'){
 51             foreach($this->getAllFilesInDir(dirname($source)) as $element){
 52                 $this->copy($element, $dest);
 53             }
 54         }
 55         else{
 56             if(! file_exists($source)){
 57                 throw new FileSystemException('Cannot copy ' . $source . ' : No such file or directory');
 58             }
 59 
 60             if(is_file($source)){
 61                 // Copy a file
 62                 if(is_dir($dest)){
 63                     $dest = $dest . self::DS . basename($source);
 64                 }
 65                 copy($source, $dest);
 66             }
 67             else{
 68                 // Copy a directory
 69                 $base = basename($source);
 70                 if(!is_dir($dest . self::DS . $base)){
 71                     mkdir($dest . self::DS . $base, fileperms($source), true);
 72                 }
 73 
 74                 // Copy all files and folder under this directory
 75                 foreach(glob($source . self::DS . '*') as $element){                
 76                     $this->copy($element, $dest . self::DS . $base);
 77                 }
 78             }
 79         }
 80     }

 83     /**
 84      * Find files by a pattern
 85      * @param string $source The directory to search in  
 86      * @param string $pattern The pattern to find the files
 87      * @param string $type The type of source to find : 'file', 'dir', 'any'     
 88      * @return array The list of files or directories found
 89      */
 90     public function find($source, $pattern, $type = self::FIND_ANY_TYPE){
 91         if(!is_dir($source)){
 92             throw new FileSystemException('The method ' . __METHOD__ . ' requires the first argument to be an existing directory : ' . $source . ' is not a directory');
 93         }
 94         switch($type){
 95             case self::FIND_FILE_ONLY :
 96                 $result = array_filter(glob($source . self::DS . $pattern), 'is_file');
 97                 break;
 98 
 99             case self::FIND_DIR_ONLY :
100                 $result = glob($source . self::DS . $pattern, GLOB_ONLYDIR);
101                 break;
102 
103             default :
104                 $result = glob($source . self::DS . $pattern);
105                 break;
106         }       
107 
108         $subdirs = glob($source . self::DS . '*', GLOB_ONLYDIR);
109         foreach($subdirs as $dir){          
110             $result = array_merge($result, self::find($dir, $pattern, $type));
111         }
112 
113         return $result;
114     }

117     /**
118      * Remove a directory or a file
119      * @param string $source The file or directory name to remove
120      * @return boolean, TRUE if the source was removed, else FALSE
121      */
122     public function remove($source){    
123         if(basename($source) == '*'){
124             foreach(self::getAllFilesInDir(dirname($source)) as $element){
125                 $this->remove($element);
126             }
127         }
128         else{   
129             if(! file_exists($source)){
130                 throw new FileSystemException('Cannot remove ' . $source . ' : No such file or directory');
131             }
132 
133             if(is_file($source)){
134                 // remove a file
135                 return unlink($source);
136             }
137             else{
138                 // remove a directory 
139                 foreach(self::getAllFilesInDir($source) as $element){
140                     $this->remove($element);                
141                 }
142                 return rmdir($source);
143             }
144         }
145     }
