 78     /**
 79      * Execute a controller method. This method is called by the router.
 80      * It execute the controller method, and triggers events before and after the method has been executed, to add widgets or other functionnalities
 81      * from another plugin than the controller's one.
 82      * @param string $method The method to execute
 83      * @param mixed The result of the controller method execution
 84      */
 85     public function compute($method){
 86         /*** Load widgets before calling the controller method ***/
 87         $class = $this->getClassname();
 88 
 89         $event = new Event($this->_plugin . '.' . $class . '.' . $method . '.' . self::BEFORE_ACTION, array('controller' => $this));
 90         $event->trigger();
 91 
 92 
 93         /*** Call the controller method ***/
 94         $args = array_splice(func_get_args(), 1);
 95         $result = call_user_func_array(array($this, $method), $args);
 96         if(App::response()->getContentType() == 'html'){
 97             // Create a phpQuery object to be modified by event listeners (widgets)
 98             $result = \phpQuery::newDocument($result);
 99         }
100 
101         /*** Load the widgets after calling the controller method ***/
102         $event = new Event($this->_plugin . '.' . $class . '.' . $method . '.' . self::AFTER_ACTION, array('controller' => $this, 'result' => $result));
103         $event->trigger();
104 
105         $result = $event->getData('result');
106         if( $result instanceof \phpQuery){
107             return $result->htmlOuter();
108         }
109         else{
110             return $result;
111         }
112     }

115     /**
116      * Add static content at the end of the DOM
117      * @param string $content The content to add
118      */
119     private function addContentAtEnd($content){
120         $action = App::router()->getCurrentAction();
121         list($tmp, $method) = explode('.', $action);
122 
123         Event::on($this->_plugin . '.' . $this->getClassname() . '.' . $method . '.' . self::AFTER_ACTION, function($event) use($content){
124             if(App::response()->getContentType() === 'html'){
125                 $dom = $event->getData('result');
126                 if($dom->find('body')->length){
127                     $dom->find('body')->append($content);
128                 }
129                 else{
130                     $dom->find("*:first")->parent()->children()->slice(-1)->after($content);
131                 }
132             }
133         });
