299     /**
300      * Check the submitted value of the input
301      * @param Form &$form The form the input is associated with
302      * @return bool True if the submitted value is valid, else false
303      */
304     public function check(&$form = null){               
305         if(empty($this->errorAt)){
306             $this->errorAt = $this->name;
307         }
308         
309         // Check, if the field is required, that a value was submitted
310         if(!empty($this->required) && ((string)$this->value == '' || $this->emptyValue && $this->value === $this->emptyValue)){
311             // The field is required but not filled
312             $form && $form->error($this->errorAt, Lang::get('form.required-field'));
313             return false;
314         }
315 
316         // Check the format of the field is correct
317         if(!empty($this->value) && !empty($this->pattern)){
318             // test the format of the field
319             if(!preg_match($this->pattern, $this->value)){
320                 // the format of the field value is not correct             
321                 $form && $form->error($this->errorAt, isset($this->errorMessage) ? $this->errorMessage : (Lang::exists('form.'.static::TYPE."-format") ? Lang::get('form.'.static::TYPE."-format") : Lang::get('form.field-format')) );
322                 return false;
323             }
324         }
325         
326         // If the value of this field must be unique in the database, check this unicity is not compromised
327         if(!empty($this->value) && $this->unique && $form ){
328             $example = new DBExample(array(
329                 '$not' => $form->reference,
330                 array($this->name => $this->dbvalue())
331             ));
332 
333             $model = $form->model;
334             if($model::getByExample($example)){
335                 // The field must be unique but is not
336                 $form->error($this->errorAt, Lang::get('form.unique-field'));
337                 return false;
338             }
339         }
340         
341         // Check custom validators
342         if(!empty($this->validators)){
343             foreach($this->validators as $validator){
344                 $error = '';
345                 if(is_callable($validator) && !$validator($this, $error)){
346                     $form->error($this->errorAt, $error);
347                     return false;
348                 }
349             }
350         }
351         
352         // The field is correctly filled (for the common checks)
353         return true;
354     }
