145     /**
146      * Get the plugin contaning a given filename
147      * @param string $file The filename to get the containing plugin of
148      * @return  Plugin The found plugin
149      */
150     public static function getFilePlugin($file){
151         // Search plugin in cache
152         if(isset(self::$filePlugins[$file])){
153             return self::$filePlugins[$file];
154         }
155 
156         if(strpos($file, PLUGINS_DIR) !== false){
157             $dir = str_replace(PLUGINS_DIR, '', $file);
158         }
159         elseif(strpos($file, MAIN_PLUGINS_DIR) !== false){
160             $dir = str_replace(MAIN_PLUGINS_DIR, '', $file);
161         }
162         else{
163             return null;
164         }
165         list($name) = explode(DIRECTORY_SEPARATOR, $dir);
166 
167         // instanciate the plugin
168         $plugin = self::get($name);
169 
170         // Register the pluygin in the memory cache
171         self::$filePlugins[$file] = &$plugin;
172 
173         // return the plugin instance
174         return $plugin;
175     }

178     /**
179       * get all the plugins
180       * @param bool $includeMain If true, include main plugins to the returned list
181       * @param bool $loadConf If set to true, load the plugins conf in the database
182       * @return array The list of plugin instances
183       */
184     public static function getAll($includeMain = true, $loadConf = false){
185         $plugins = array();
186         $dirs = $includeMain ? array(MAIN_PLUGINS_DIR, PLUGINS_DIR) : array(PLUGINS_DIR);
187 
188         if($loadConf && App::conf()->has('db')){
189             $configs = App::db()->select(array(
190                 'from' => DB::getFullTablename(self::TABLE),
191                 'index' => 'name',
192                 'return' => DB::RETURN_OBJECT
193             ));
194         }
195         else{
196             $configs = array();
197         }
198 
199         foreach($dirs as $dir){
200             foreach(glob($dir . '*', GLOB_ONLYDIR) as $dir){
201                 $name = basename($dir);
202                 $config = isset($configs[$name]) ? $configs[$name] : null;
203 
204                 $plugin = self::get($name);
205                 if(!$plugin->isMainPlugin()){
206                     $plugin->active = isset($config->active) ? $config->active : false;
207                 }
208                 $plugins[$name] = $plugin;
209             }
210         }
211 
212         return $plugins;
213     }

391             return $baseUrl;

393         else{
394             $privateFilename = $this->getStaticDir() . $basename;
395             $publicFilename = $this->getPublicStaticDir() . $basename;
396 
397             if(is_file($privateFilename) && (!is_file($publicFilename) || filemtime($privateFilename) > filemtime($publicFilename))){
398                 if(!is_dir(dirname($publicFilename))){
399                     mkdir(dirname($publicFilename), 0755, true);
400                 }
401 
402                 copy($privateFilename, $publicFilename);
403             }
404

406         }
